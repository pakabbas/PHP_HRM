-- =========================================
-- USERS & EMPLOYEES
-- =========================================

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NULL,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role ENUM('admin','hr','manager','employee') DEFAULT 'employee',
    status TINYINT(1) DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE employees (
    id INT AUTO_INCREMENT PRIMARY KEY,
    emp_code VARCHAR(50) UNIQUE NOT NULL,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100),
    gender ENUM('male','female','other'),
    dob DATE,
    cnic VARCHAR(30),
    phone VARCHAR(30),
    email VARCHAR(100),
    address VARCHAR(255),
    city_id INT,
    department_id INT,
    designation_id INT,
    joining_date DATE,
    leaving_date DATE NULL,
    status ENUM('active','inactive','terminated','resigned') DEFAULT 'active',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- =========================================
-- CONFIG / DICTIONARY TABLES
-- =========================================

CREATE TABLE departments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    department_name VARCHAR(100) UNIQUE,
    status TINYINT(1) DEFAULT 1
);

CREATE TABLE designations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    designation_name VARCHAR(100),
    status TINYINT(1) DEFAULT 1
);

CREATE TABLE cities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    city_name VARCHAR(100) UNIQUE,
    status TINYINT(1) DEFAULT 1
);

CREATE TABLE configurations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    config_type VARCHAR(100) NOT NULL,
    config_key VARCHAR(100) NOT NULL,
    config_value VARCHAR(255) NOT NULL,
    status TINYINT(1) DEFAULT 1
);


-- =========================================
-- ATTENDANCE & HOLIDAYS
-- =========================================

CREATE TABLE attendance (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    attendance_date DATE NOT NULL,
    check_in TIME,
    check_out TIME,
    status ENUM('present','absent','leave','half_day','holiday') DEFAULT 'present',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE holidays (
    id INT AUTO_INCREMENT PRIMARY KEY,
    holiday_date DATE NOT NULL,
    description VARCHAR(150)
);


-- =========================================
-- LEAVE MANAGEMENT
-- =========================================

CREATE TABLE leave_types (
    id INT AUTO_INCREMENT PRIMARY KEY,
    type_name VARCHAR(100) NOT NULL,
    default_days INT DEFAULT 0,
    carry_forward TINYINT(1) DEFAULT 1,
    status TINYINT(1) DEFAULT 1
);

CREATE TABLE leave_requests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    leave_type_id INT NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    total_days INT,
    status ENUM('pending','approved','rejected','cancelled') DEFAULT 'pending',
    remarks VARCHAR(255),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE leave_balances (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    leave_type_id INT NOT NULL,
    allocated INT DEFAULT 0,
    used INT DEFAULT 0,
    remaining INT GENERATED ALWAYS AS (allocated - used) STORED,
    year INT NOT NULL
);


-- =========================================
-- PAYROLL & SALARY MANAGEMENT
-- =========================================

CREATE TABLE employee_salary_settings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    basic_salary DECIMAL(10,2) NOT NULL,
    overtime_rate DECIMAL(10,2) DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE salary_allowances (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    title VARCHAR(100),
    amount DECIMAL(10,2) DEFAULT 0
);

CREATE TABLE salary_deductions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    title VARCHAR(100),
    amount DECIMAL(10,2) DEFAULT 0
);

CREATE TABLE overtime (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    hours DECIMAL(5,2),
    rate DECIMAL(10,2),
    amount DECIMAL(10,2),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE loans (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    amount DECIMAL(10,2),
    remaining DECIMAL(10,2),
    monthly_installment DECIMAL(10,2),
    status ENUM('active','closed') DEFAULT 'active',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE payroll_runs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    month INT NOT NULL,
    year INT NOT NULL,
    gross_salary DECIMAL(10,2),
    total_allowances DECIMAL(10,2),
    total_deductions DECIMAL(10,2),
    overtime_amount DECIMAL(10,2),
    unpaid_leave_deductions DECIMAL(10,2),
    loan_deduction DECIMAL(10,2),
    net_salary DECIMAL(10,2),
    status ENUM('draft','processed','paid') DEFAULT 'draft',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- REPORT HISTORY

CREATE TABLE reports (
    id INT AUTO_INCREMENT PRIMARY KEY,
    report_name VARCHAR(150),
    report_type VARCHAR(100),
    created_by INT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
